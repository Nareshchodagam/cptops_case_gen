- set your DC env
Exec: DC=wax

- create a local remote_transfer directory
Exec: mkdir -p ~/releaserunner/remote_transfer

- pull necessary utility scripts
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/nagios_monitor.py ~/nagios_monitor.py
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/common.py ~/common.py
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/idbhost.py ~/idbhost.py
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/notIPMI.sh ~/notIPMI.sh
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/check_reconnect.py ~/check_reconnect.py
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/validate_rh.py ~/releaserunner/remote_transfer/validate_rh.py
Exec: svn export svn://vc-wax/subversion/tools/automation/scripts/opseng/computedeploy/patching/bin/common.py ~/releaserunner/remote_transfer/common.py
Exec: svn export svn://vc-wax/subversion/jumpstart/common/solaris-and-linux/sysfiles/system_update.sh ~/releaserunner/remote_transfer/system_update.sh


RR based commands

-copy it up to the target host
release_runner.pl -forced_host cs32-search1-1-wax -force_update_bootstrap -c sudo_cmd -m "ls" -threads -auto2

-Validate the host needs to be patched
release_runner.pl -forced_host cs32-search1-1-wax -c sudo_cmd -m "./remote_transfer/validate_rh.py -k 2.6.32-504.12.2.el6.x86_64 -r 6.6 -u && echo 'already patched'" -threads -auto2

-test boot device
./notIPMI.sh power_status cs32-search1-1-wax admin HDD ANY
 
-disable monitoring on appropriate monitor host
Exec: /opt/rh/python27/root/usr/bin/python2.7 ~/nagios_monitor.py -H cs32-search1-1-wax -c disable

-server status
release_runner.pl -invdb_mode -superpod SPX -cluster CS32 -product sfdc-base,sfdc-splunk-forwarder -host cs32-search1-1-wax -c server_status -auto2

-Test the  update script:
release_runner.pl -forced_host cs32-search1-1-wax -c sudo_cmd -m "./remote_transfer/system_update.sh -t" -threads -auto2

-check version
release_runner.pl -forced_host cs32-search1-1-wax -c sudo_cmd -m "uname -a && cat /etc/redhatrelease" -threads -auto2

- Wait for host to boot up
Exec: ~/check_reconnect.py -H cs32-search1-1-wax

- Validate patching
release_runner.pl -forced_host cs32-search1-1-wax -c sudo_cmd -m "./remote_transfer/validate_rh.py -k 2.6.32-504.12.2.el6.x86_64 -r 6.6 -b 300 -u" -threads -auto2

-test app
release_runner.pl -invdb_mode -superpod SPX -cluster CS32 -product  sfdc-base,sfdc-splunk-forwarder  -device_role search -host cs32-search1-1-wax -c server_status -auto2

- add to pool
release_runner.pl -invdb_mode -superpod SPX -cluster CS32 -product sfdc-base -device_role search -host cs32-search1-1-wax -c pingcheck -auto2

-test enable monitoring on appropriate monitor host
Exec: /opt/rh/python27/root/usr/bin/python2.7 ~/nagios_monitor.py -H cs32-search1-1-wax -c enable
